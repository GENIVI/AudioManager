332  {
333  timespec result;
334 
335  if ((a.tv_sec < b.tv_sec) || ((a.tv_sec == b.tv_sec) && (a.tv_nsec <= b.tv_nsec)))
336  {
337  result.tv_sec = result.tv_nsec = 0;
338  }
